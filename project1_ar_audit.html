SQL — Anomaly Detection Query
Core query used to surface unapplied credits and unallocated funds
-- Identify accounts with credits exceeding invoiced amounts
-- or funds unapplied for more than 90 days

SELECT
    a.account_id,
    a.account_type,
    a.company_name,
    SUM(inv.invoice_amount)                        AS total_invoiced,
    SUM(pay.payment_amount)                        AS total_paid,
    SUM(cm.credit_amount)                          AS total_credits,
    (SUM(pay.payment_amount) - SUM(inv.invoice_amount)) AS net_balance,
    DATEDIFF(CURDATE(), MAX(cm.credit_date))          AS days_unapplied,

    CASE
        WHEN cm.credit_amount > 0 AND cm.applied_flag = 0
             AND DATEDIFF(CURDATE(), cm.credit_date) > 90
             THEN 'UNAPPLIED_CREDIT'
        WHEN (SUM(pay.payment_amount) - SUM(inv.invoice_amount)) > 1000
             THEN 'UNALLOCATED_FUND'
        WHEN COUNT(pay.payment_id) > COUNT(DISTINCT inv.invoice_id)
             THEN 'POSSIBLE_DUPLICATE'
        ELSE 'REVIEW'
    END AS anomaly_type

FROM accounts a
JOIN invoices   inv ON a.account_id = inv.account_id
JOIN payments   pay ON a.account_id = pay.account_id
LEFT JOIN credit_memos cm  ON a.account_id = cm.account_id

WHERE inv.invoice_date >= DATE_SUB(CURDATE(), INTERVAL 24 MONTH)

GROUP BY
    a.account_id, a.account_type, a.company_name, cm.credit_amount,
    cm.applied_flag, cm.credit_date

HAVING
    net_balance > 500
    OR (cm.credit_amount > 0 AND days_unapplied > 90)

ORDER BY net_balance DESC;
